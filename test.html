class Particle():
    def __init__(self, Nx, Ny, lifeMin, lifeMax):
        self.Nx = Nx
        self.Ny = Ny
        self.lifeMin = lifeMin
        self.lifeMax = lifeMax
        self.Spawn()
        self.positions = []

    def Spawn(self):
        self.x = self.Nx * random.random()
        self.y = self.Ny * random.random()
        self.life = 0
        self.lifeTime = (self.lifeMax * random.random()) + self.lifeMin 

    def Update(self, amountMovedX, amountMovedY, n):
        self.x += amountMovedX
        self.y += amountMovedY
        self.life += 1

        if (self.life > self.lifeTime):
            self.Spawn()

        if (self.x > Nx or self.x < 0):
            self.Spawn()

        if (self.y > Ny or self.y < 0):
            self.Spawn()

        self.TrackPosition(n)

    def TrackPosition(self, n):
        data = PositionData(self.x, self.y, n)
        self.positions.append(data)

class PositionData():
    def __init__(self, x, y, n):
        self.x = x
        self.y = y
        self.n = n
